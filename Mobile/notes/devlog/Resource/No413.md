export async function parsePDFRecipes(filePath: string): Promise<ParseResult> {

/**
 * Parse individual recipe page
 */
function parseRecipePage(text: string, name: string, pageNum: number): ParsedRecipe {
  const recipe: ParsedRecipe = {
    id: crypto.randomUUID(),
    name: name.trim(),
    name_en: name.trim(),
    servings: 8,
    bake_time_minutes: 0,
    ingredients: [],
    directions: [],
    page_number: pageNum,
  };

  const lines = text.split('\n');
  let inIngredients = false;
  let inDirections = false;

  for (const line of lines) {
    const trimmed = line.trim();

    // Detect servings
    if (/serving|portion|porcion|person|people|–Ω–Ψ―Ä―Ü–Η―è|–Ω–Β―Ä―¹–Ψ–Ϋ–Α/i.test(trimmed)) {

      inDirections = false;

    // Detect cook time
    if (/time|–≤―Ä–Β–Φ–Β|cook|min|minute|prep|preparation|–Ω―Ä–Η–≥–Ψ―²–≤/i.test(trimmed)) {
      const match = trimmed.match(/(\d+)/);

    // Detect ingredients section
    if (/ingredient|―¹―ä―¹―²–Α–≤–Κ|product|–Ω―Ä–Ψ–¥―É–Κ―²|–Η–Ϋ–≥―Ä–Β–¥–Η–Β–Ϋ―²/i.test(trimmed)) {
      inIngredients = true;
      inDirections = false;
      continue;
    }

    // Detect directions section
    if (/direction|–Η–Ϋ―¹―²―Ä―É–Κ―Ü|–Ω―Ä–Η–≥–Ψ―²–≤|step|―¹―²―ä–Ω–Κ|method|–Φ–Β―²–Ψ–¥–Η|process/i.test(trimmed)) {

      inDirections = true;
      continue;

    }

    }

}
